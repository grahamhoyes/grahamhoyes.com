---
title: Cranberry Sauce
date: 2025-04-19
description:
    Slightly tart cranberry sauce with orange zest
categories: ["side"]
---

import { Ingredients, Steps } from "components/recipes";

<Ingredients>
    - One 340g bag of cranberries
    - 150g white sugar
    - One large orange, zested and juiced
    - 100 mL water
</Ingredients>

<Steps>
    1. Zest and juice the orange
    2. Combine all ingredients in a small saucepan
    3. Bring to a boil over medium heat
    4. Reduce heat and simmer for 5-10 minutes, until the cranberries burst and the sauce thickens
</Steps>
